Go语言Linux网络编程 使用原始套接字

2020-10-26
网络编程 GO语言

同时监听TCP和UDP

面试常见题，是否可以在一个linux的网络端口同时处理TCP和UDP。
常间的同时处理TCP和UDP的程序有超级服务inetd 和安卓的调试服务 adbd。

一个套接字只能绑定一个端口，如果一个端口要同时处理两种服务，就要申请两个套接字来监听一个端口。一个流式套接字，一个数据报套接字。

示例

package main


import (
	"fmt"
	"io"
	"log"
	"net"
	"os"
)

// 错误处理
func doErr(err error) {
	if err != nil {
		log.Println(err)
		os.Exit(1)
	}
}

// 处理tcp
func handle(conn net.Conn) {
	defer conn.Close()
	var buf [125]byte
	for {
		n, err := conn.Read(buf[:])
		if err == io.EOF {
			break
		} else {
			doErr(err)
		}

		_, err = conn.Write(buf[:n])
		doErr(err)
	}
}

func main() {
  // tcp listen
	tcpListen, err := net.Listen("tcp", ":8086")
	doErr(err)
	defer tcpListen.Close()

  // udp listen
	udpAddr, _ := net.ResolveUDPAddr("udp", ":8086")
	var udpListen *net.UDPConn
	udpListen, err = net.ListenUDP("udp", udpAddr)
	doErr(err)
	defer udpListen.Close()

  // 处理 udp 
	go func() {
		for {
			var buf [125]byte
			n, addr, err := udpListen.ReadFromUDP(buf[:])
			doErr(err)
			_, err = udpListen.WriteToUDP(buf[:n], addr)
			doErr(err)
		}
	}()

  // 处理 tcp
	for {
		conn, _ := tcpListen.Accept()
		go handle(conn)
	}
}

测试： $ nc 0.0.0.0 8086 # tcp测试
$ echo "some message" | nc -4u 0.0.0.0 8086 # upd测试
